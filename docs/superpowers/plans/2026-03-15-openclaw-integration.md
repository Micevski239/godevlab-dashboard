# OpenClaw AI Integration — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a full OpenClaw AI section to the GoGevgelija Dashboard with chat, automations, and logs pages — admin-only, powered by direct WebSocket to local OpenClaw gateway.

**Architecture:** Direct WebSocket from browser to local OpenClaw gateway (ws://localhost:18789). Ephemeral token exchange via server-side API route. React Context provides singleton gateway instance to all OpenClaw pages. No new Supabase tables — OpenClaw manages its own state.

**Tech Stack:** Next.js 15 App Router, React 19, TypeScript, Tailwind CSS, shadcn/ui, React Query, WebSocket (browser-native), react-markdown

```typescript
// src/lib/openclaw/gateway.ts
import type {
  OpenClawConnectionState,
  OpenClawSession,
  OpenClawMessage,
  OpenClawAutomation,
  OpenClawLogEntry,
  OpenClawLogStats,
  OpenClawGatewayEvent,
} from "@/types/openclaw";

type EventHandler = (event: OpenClawGatewayEvent) => void;

interface GatewayOptions {
  url: string;
  token: string;
  onStateChange?: (state: OpenClawConnectionState) => void;
}

export class OpenClawGateway {
  private ws: WebSocket | null = null;
  private url: string;
  private token: string;
  private reconnectAttempts = 0;
  private maxReconnectDelay = 30000;
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;
  private listeners: Set<EventHandler> = new Set();
  private onStateChange?: (state: OpenClawConnectionState) => void;
  private _state: OpenClawConnectionState = "disconnected";
  private abortControllers: Map<string, AbortController> = new Map();

  constructor(options: GatewayOptions) {
    this.url = options.url;
    this.token = options.token;
    this.onStateChange = options.onStateChange;
  }

  get state(): OpenClawConnectionState {
    return this._state;
  }

  private setState(state: OpenClawConnectionState) {
    this._state = state;
    this.onStateChange?.(state);
  }

  connect() {
    if (this.ws?.readyState === WebSocket.OPEN) return;

    this.setState("connecting");
    const wsUrl = `${this.url}?token=${encodeURIComponent(this.token)}`;

    try {
      this.ws = new WebSocket(wsUrl);
    } catch {
      this.setState("error");
      this.scheduleReconnect();
      return;
    }

    this.ws.onopen = () => {
      this.reconnectAttempts = 0;
      this.setState("connected");
      this.emit({ type: "connected" });
    };

    this.ws.onclose = () => {
      this.setState("disconnected");
      this.emit({ type: "disconnected" });
      this.scheduleReconnect();
    };

    this.ws.onerror = () => {
      this.setState("error");
      this.emit({ type: "error", data: "WebSocket connection error" });
    };

    this.ws.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data);
        this.handleMessage(data);
      } catch {
        this.emit({
          type: "error",
          data: "Failed to parse gateway message",
        });
      }
    };
  }

  disconnect() {
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
      this.reconnectTimer = null;
    }
    this.abortControllers.forEach((controller) => controller.abort());
    this.abortControllers.clear();
    if (this.ws) {
      this.ws.onclose = null; // prevent reconnect
      this.ws.close();
      this.ws = null;
    }
    this.setState("disconnected");
  }

  updateToken(token: string) {
    this.token = token;
    // Reconnect with new token if currently connected
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.onclose = null;
      this.ws.close();
      this.ws = null;
      this.connect();
    }
  }

  // --- Event system ---

  on(handler: EventHandler) {
    this.listeners.add(handler);
    return () => this.listeners.delete(handler);
  }

  private emit(event: OpenClawGatewayEvent) {
    this.listeners.forEach((handler) => handler(event));
  }

  private handleMessage(data: Record<string, unknown>) {
    // Validate expected message shape
    if (!data || typeof data !== "object" || !("type" in data)) {
      this.emit({
        type: "error",
        data: "Malformed gateway message: missing type field",
      });
      return;
    }

    if (data.type === "token") {
      this.emit({ type: "token", data: data.content });
    } else if (data.type === "message") {
      this.emit({ type: "message", data });
    } else if (data.type === "tool_call") {
      this.emit({ type: "tool_call", data });
    } else if (data.type === "error") {
      this.emit({ type: "error", data });
    }
  }

  // --- Reconnect ---

  private scheduleReconnect() {
    if (this.reconnectTimer) return;
    const delay = Math.min(
      1000 * Math.pow(2, this.reconnectAttempts),
      this.maxReconnectDelay
    );
    this.reconnectAttempts++;
    this.reconnectTimer = setTimeout(() => {
      this.reconnectTimer = null;
      this.connect();
    }, delay);
  }

  reconnect() {
    this.reconnectAttempts = 0;
    this.disconnect();
    this.connect();
  }

  // --- API Methods ---

  private send(payload: Record<string, unknown>) {
    if (this.ws?.readyState !== WebSocket.OPEN) {
      throw new Error("Gateway not connected");
    }
    this.ws.send(JSON.stringify(payload));
  }

  async sendMessage(
    sessionId: string,
    text: string,
    onToken: (token: string) => void,
    onComplete: (message: OpenClawMessage) => void,
    onError: (error: string) => void
  ): Promise<AbortController> {
    const controller = new AbortController();
    const requestId = `msg-${Date.now()}`;
    this.abortControllers.set(requestId, controller);

    const cleanup = this.on((event) => {
      if (controller.signal.aborted) {
        cleanup();
        this.abortControllers.delete(requestId);
        return;
      }
      if (event.type === "token") {
        onToken(event.data as string);
      } else if (event.type === "message") {
        onComplete(event.data as OpenClawMessage);
        cleanup();
        this.abortControllers.delete(requestId);
      } else if (event.type === "error") {
        onError(event.data as string);
        cleanup();
        this.abortControllers.delete(requestId);
      }
    });

    controller.signal.addEventListener("abort", () => {
      this.send({ type: "abort", requestId });
      cleanup();
      this.abortControllers.delete(requestId);
    });

    this.send({
      type: "message",
      sessionId,
      content: text,
      requestId,
    });

    return controller;
  }

  async listSessions(): Promise<OpenClawSession[]> {
    return this.request<OpenClawSession[]>({ type: "list_sessions" });
  }

  async createSession(): Promise<OpenClawSession> {
    return this.request<OpenClawSession>({ type: "create_session" });
  }

  async deleteSession(sessionId: string): Promise<void> {
    await this.request({ type: "delete_session", sessionId });
  }

  async listCronJobs(): Promise<OpenClawAutomation[]> {
    return this.request<OpenClawAutomation[]>({ type: "list_cron_jobs" });
  }

  async updateCronJob(
    id: string,
    update: { status: "active" | "paused" }
  ): Promise<OpenClawAutomation> {
    return this.request<OpenClawAutomation>({
      type: "update_cron_job",
      id,
      ...update,
    });
  }

  async getMessages(sessionId: string): Promise<OpenClawMessage[]> {
    return this.request<OpenClawMessage[]>({
      type: "get_messages",
      sessionId,
    });
  }

  async getTaskHistory(filters?: {
    type?: string;
    since?: string;
  }): Promise<{ entries: OpenClawLogEntry[]; stats: OpenClawLogStats }> {
    return this.request({
      type: "get_task_history",
      ...filters,
    });
  }

  // Generic request-response over WebSocket
  private request<T>(payload: Record<string, unknown>): Promise<T> {
    return new Promise((resolve, reject) => {
      const requestId = `req-${Date.now()}-${Math.random().toString(36).slice(2)}`;
      const timeout = setTimeout(() => {
        cleanup();
        reject(new Error("Gateway request timed out"));
      }, 10000);

      const cleanup = this.on((event) => {
        if (event.type === "message") {
          const data = event.data as Record<string, unknown>;
          if (data.requestId === requestId) {
            clearTimeout(timeout);
            cleanup();
            resolve(data.result as T);
          }
        } else if (event.type === "error") {
          const data = event.data as Record<string, unknown>;
          if (
            typeof data === "object" &&
            data !== null &&
            "requestId" in data &&
            data.requestId === requestId
          ) {
            clearTimeout(timeout);
            cleanup();
            reject(new Error((data.message as string) || "Gateway error"));
          }
        }
      });

      this.send({ ...payload, requestId });
    });
  }
}
```

- [ ] **Step 1: Create the token route**

```typescript
// src/app/api/openclaw/token/route.ts
import { NextResponse } from "next/server";
import { createClient } from "@/lib/supabase/server";

export async function POST() {
  try {
    const supabase = await createClient();
    const {
      data: { user },
      error,
    } = await supabase.auth.getUser();

    if (error || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const { data: employee } = await supabase
      .from("employees")
      .select("role")
      .eq("id", user.id)
      .single();

    if (employee?.role !== "admin") {
      return NextResponse.json({ error: "Forbidden" }, { status: 403 });
    }

    const gatewayToken = process.env.OPENCLAW_GATEWAY_TOKEN;
    if (!gatewayToken) {
      return NextResponse.json(
        { error: "Gateway token not configured" },
        { status: 500 }
      );
    }

    const res = await fetch("http://localhost:18789/api/v1/token", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${gatewayToken}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ ttl: 300 }), // 5 minutes
    });

    if (!res.ok) {
      return NextResponse.json(
        { error: "Gateway unavailable" },
        { status: 502 }
      );
    }

    const { token } = await res.json();
    return NextResponse.json({ token });
  } catch {
    return NextResponse.json(
      { error: "Failed to obtain gateway token" },
      { status: 500 }
    );
  }
}
```

```typescript
// src/components/openclaw/provider.tsx
"use client";

import {
  createContext,
  useContext,
  useEffect,
  useRef,
  useState,
  useCallback,
  type ReactNode,
} from "react";
import { OpenClawGateway } from "@/lib/openclaw/gateway";
import type { OpenClawConnectionState } from "@/types/openclaw";

interface OpenClawContextValue {
  gateway: OpenClawGateway | null;
  connectionState: OpenClawConnectionState;
  isConnected: boolean;
  error: string | null;
  reconnect: () => void;
}

const OpenClawContext = createContext<OpenClawContextValue | null>(null);

const TOKEN_REFRESH_INTERVAL = 4 * 60 * 1000; // 4 minutes

export function OpenClawProvider({ children }: { children: ReactNode }) {
  const gatewayRef = useRef<OpenClawGateway | null>(null);
  const [gatewayInstance, setGatewayInstance] = useState<OpenClawGateway | null>(null);
  const refreshTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);
  const [connectionState, setConnectionState] =
    useState<OpenClawConnectionState>("connecting");
  const [error, setError] = useState<string | null>(null);

  const fetchToken = useCallback(async (): Promise<string | null> => {
    try {
      const res = await fetch("/api/openclaw/token", { method: "POST" });
      if (!res.ok) {
        const data = await res.json();
        setError(data.error || "Failed to fetch token");
        return null;
      }
      const { token } = await res.json();
      setError(null);
      return token;
    } catch {
      setError("Failed to reach token endpoint");
      return null;
    }
  }, []);

  const initGateway = useCallback(async () => {
    const token = await fetchToken();
    if (!token) {
      setConnectionState("error");
      return;
    }

    const url = process.env.NEXT_PUBLIC_OPENCLAW_GATEWAY_URL || "ws://localhost:18789";
    const gw = new OpenClawGateway({
      url,
      token,
      onStateChange: setConnectionState,
    });

    gw.on((event) => {
      if (event.type === "error") {
        setError(event.data as string);
      }
    });

    gatewayRef.current = gw;
    setGatewayInstance(gw);
    gw.connect();

    // Refresh token periodically
    refreshTimerRef.current = setInterval(async () => {
      const newToken = await fetchToken();
      if (newToken && gatewayRef.current) {
        gatewayRef.current.updateToken(newToken);
      }
    }, TOKEN_REFRESH_INTERVAL);
  }, [fetchToken]);

  useEffect(() => {
    initGateway();

    return () => {
      gatewayRef.current?.disconnect();
      gatewayRef.current = null;
      if (refreshTimerRef.current) {
        clearInterval(refreshTimerRef.current);
        refreshTimerRef.current = null;
      }
    };
  }, [initGateway]);

  // Reconnect on window focus
  useEffect(() => {
    const handleFocus = () => {
      if (gatewayRef.current?.state === "disconnected") {
        gatewayRef.current.reconnect();
      }
    };
    window.addEventListener("focus", handleFocus);
    return () => window.removeEventListener("focus", handleFocus);
  }, []);

  const reconnect = useCallback(() => {
    gatewayRef.current?.reconnect();
  }, []);

  return (
    <OpenClawContext.Provider
      value={{
        gateway: gatewayInstance,
        connectionState,
        isConnected: connectionState === "connected",
        error,
        reconnect,
      }}
    >
      {children}
    </OpenClawContext.Provider>
  );
}

export function useOpenClawContext() {
  const context = useContext(OpenClawContext);
  if (!context) {
    throw new Error("useOpenClawContext must be used within OpenClawProvider");
  }
  return context;
}
```

```typescript
"use client";

import { useCallback, useState } from "react";
import { useQuery, useQueryClient } from "@tanstack/react-query";
import { useOpenClaw } from "./use-openclaw";
import type { OpenClawSession } from "@/types/openclaw";

export function useOpenClawSessions() {
  const { gateway, isConnected } = useOpenClaw();
  const queryClient = useQueryClient();
  const [activeSessionId, setActiveSessionId] = useState<string | null>(null);

  const { data: sessions = [], isLoading } = useQuery<OpenClawSession[]>({
    queryKey: ["openclaw-sessions"],
    queryFn: async () => {
      if (!gateway) return [];
      return gateway.listSessions();
    },
    enabled: isConnected && !!gateway,
    refetchInterval: 30000,
  });

  const createSession = useCallback(async () => {
    if (!gateway) return null;
    const session = await gateway.createSession();
    queryClient.invalidateQueries({ queryKey: ["openclaw-sessions"] });
    setActiveSessionId(session.id);
    return session;
  }, [gateway, queryClient]);

  const deleteSession = useCallback(
    async (sessionId: string) => {
      if (!gateway) return;
      await gateway.deleteSession(sessionId);
      queryClient.invalidateQueries({ queryKey: ["openclaw-sessions"] });
      if (activeSessionId === sessionId) {
        setActiveSessionId(null);
      }
    },
    [gateway, queryClient, activeSessionId]
  );

  const switchSession = useCallback((sessionId: string) => {
    setActiveSessionId(sessionId);
  }, []);

  return {
    sessions,
    activeSessionId,
    createSession,
    deleteSession,
    switchSession,
    isLoading,
  };
}
```

```typescript
import { useCallback, useEffect, useRef, useState } from "react";
import { useOpenClaw } from "./use-openclaw";

export function useOpenClawChat(sessionId: string | null) {

  const [messages, setMessages] = useState<OpenClawMessage[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const abortRef = useRef<AbortController | null>(null);
  const streamingContentRef = useRef("");

  // Load message history when session changes
  useEffect(() => {
    setMessages([]);
    if (!sessionId || !gateway) return;
    gateway.getMessages(sessionId).then(setMessages).catch(() => {
      // Session may be new with no messages
    });
  }, [sessionId, gateway]);

  const sendMessage = useCallback(
    async (text: string, overrideSessionId?: string) => {
      const targetSessionId = overrideSessionId || sessionId;
      if (!gateway || !targetSessionId) return;

      // Add user message
      const userMessage: OpenClawMessage = {
        id: `user-${Date.now()}`,
        role: "user",
        content: text,
        timestamp: new Date().toISOString(),
      };
      setMessages((prev) => [...prev, userMessage]);
      setIsStreaming(true);
      streamingContentRef.current = "";

      // Add placeholder assistant message
      const assistantId = `assistant-${Date.now()}`;
      setMessages((prev) => [
        ...prev,
        {
          id: assistantId,
          role: "assistant",
          content: "",
          timestamp: new Date().toISOString(),
        },
      ]);

      try {
        const controller = await gateway.sendMessage(
          targetSessionId,
          text,
          // onToken
          (token: string) => {
            streamingContentRef.current += token;
            setMessages((prev) =>
              prev.map((msg) =>
                msg.id === assistantId
                  ? { ...msg, content: streamingContentRef.current }
                  : msg
              )
            );
          },
          // onComplete
          (completeMessage: OpenClawMessage) => {
            setMessages((prev) =>
              prev.map((msg) =>
                msg.id === assistantId ? { ...completeMessage, id: assistantId } : msg
              )
            );
            setIsStreaming(false);
          },
          // onError
          (error: string) => {
            setMessages((prev) =>
              prev.map((msg) =>
                msg.id === assistantId
                  ? { ...msg, content: `Error: ${error}` }
                  : msg
              )
            );
            setIsStreaming(false);
          }
        );
        abortRef.current = controller;
      } catch {
        setIsStreaming(false);
      }
    },
    [gateway, sessionId]
  );

  const abort = useCallback(() => {
    abortRef.current?.abort();
    abortRef.current = null;
    setIsStreaming(false);
  }, []);

  return { messages, isStreaming, sendMessage, abort };
}
```

```typescript
"use client";

import { useMemo } from "react";
import { useQuery } from "@tanstack/react-query";
import { useOpenClaw } from "./use-openclaw";
import type { OpenClawLogEntry, OpenClawLogStats } from "@/types/openclaw";

interface LogFilters {
  type?: string;
  dateRange?: "today" | "7d" | "30d";
}

function getDateSince(range: string): string {
  const now = new Date();
  if (range === "today") {
    now.setHours(0, 0, 0, 0);
  } else if (range === "7d") {
    now.setDate(now.getDate() - 7);
  } else if (range === "30d") {
    now.setDate(now.getDate() - 30);
  }
  return now.toISOString();
}

export function useOpenClawLogs(filters: LogFilters = {}) {
  const { gateway, isConnected } = useOpenClaw();

  const since = filters.dateRange ? getDateSince(filters.dateRange) : undefined;

  const { data, isLoading } = useQuery<{
    entries: OpenClawLogEntry[];
    stats: OpenClawLogStats;
  }>({
    queryKey: ["openclaw-logs", since],
    queryFn: async () => {
      if (!gateway) {
        return {
          entries: [],
          stats: { successful: 0, errors: 0, eventsCreated: 0, avgResponseMs: 0 },
        };
      }
      return gateway.getTaskHistory({ since });
    },
    enabled: isConnected && !!gateway,
    refetchInterval: 30000,
  });

  // Client-side type filter
  const filteredEntries = useMemo(() => {
    if (!data?.entries) return [];
    if (!filters.type || filters.type === "all") return data.entries;
    return data.entries.filter((entry) => entry.type === filters.type);
  }, [data?.entries, filters.type]);

  return {
    logs: filteredEntries,
    stats: data?.stats ?? {
      successful: 0,
      errors: 0,
      eventsCreated: 0,
      avgResponseMs: 0,
    },
    isLoading,
  };
}
```
